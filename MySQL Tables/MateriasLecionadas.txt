ORIGINAL

CREATE TABLE materias_lecionadas(
idmateria_lecionada INTEGER AUTO_INCREMENT PRIMARY KEY,
usuario VARCHAR(50) REFERENCES pessoa(usuario),
nome_materia VARCHAR(30));

ALTERADO

CREATE TABLE materias_lecionadas(
idmateria_lecionada INTEGER AUTO_INCREMENT PRIMARY KEY,
cpf VARCHAR(11) REFERENCES pessoa(cpf),
email VARCHAR(50),
nome_materia VARCHAR(30));
